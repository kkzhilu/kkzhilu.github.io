/**
 * MAP对象，实现MAP功能
 *
 * 例子：
 * var map = new JsMap();
 *
 * map.put("key", "value");
 * var val = map.get("key")
 */
function JsMap() {
    var p = {
        size: 0,
        obj: {}
    };

    this.clear = function() {
        p.size = 0;
        p.obj = {};
    };

    this.contains = function(key) {
        return p.obj[key] != null || p.obj[key] !== undefined;
    };

    this.get = function(key) {
        if (this.contains(key)) {
            return p.obj[key];
        } else {
            return null;
        }
    };

    this.put = function(key, val) {
        if (!this.contains(key)) {
            p.size++;
        }
        p.obj[key] = val;
    };

    this.size = function() {
        return p.size;
    };

    this.keys = function() {
        var arr = [];
        for (var key in p.obj) {
            if (p.obj.hasOwnProperty(key)) {
                arr.push(key);
            }
        };
        return arr;
    };

    this.values = function() {
        var values = [];
        for (var key of this.keys()) {
            values.push(this.get(key));
        }
        return values;
    };

    this.remove = function(key) {
        if (this.contains(key)) {
            p.size--;
            delete p.obj[key];
        }
    };
}

跨页处理

<script>

    // 存储CheckBox选择项
    var boxsMap = new JsMap();

    /***
     * 复选框点击事件
     */
    function changeCheck(totalBox) {
        var currBox = $(totalBox);
        $('#table_scripts table').find('td:first-child .cbx-case').each(function(index, item) {
            var box = $(item);
            if (currBox.is(':checked')) {
                //...................
                boxsMap.put(script.scriptNo, script)
            } else {
                boxsMap.remove(box.attr("data-script-no"))
            }
        });

        // 判断是否显示Tips
        showBoxTips();
    }

    /***
     * CheckBox点击事件
     */
    function clickRadio(box) {
        var currBox = $(box);
        var isCheck = currBox.is(':checked');
        if (isCheck) {
            //...................
            boxsMap.put(script.scriptNo, script)
        } else {
            boxsMap.remove(currBox.attr("data-script-no"));
        }

        // 判断是否显示Tips
        showBoxTips();
    }

    /***
     * 是否需要展示BoxTips | 已选择：XX
     */
    function showBoxTips() {
        let size = boxsMap.size();
        var $this = $('#BoxTips');
        if (size > 0) {
            $this.css("visibility","visible");
            $this.html("已选择数量：" + size);
        } else {
            $this.css("visibility","hidden");
        }
    }

    /***
     * 回显数据
     */
    function returnCheck() {
        var box  = $("input[name='script-boxs']");
        box.each(function(){
            var key = $(this).attr("data-script-no");
            if (boxsMap.contains(key)) {
                $(this).attr("checked", "checked");
            }
        });

        // 回显时处理 -> caseBoxAll
        var length = $("input[name='script-boxs']:checked").length;
        if (box.length > 0 && box.length === length) {
            $('#caseBoxAll').attr("checked", "checked");
        }
    }

    /**
     * 初始化CheckBox
     */
    function initJsMap() {
        boxsMap.clear();

        showBoxTips();

        // 重新渲染CheckBox
        var totalCheck = document.getElementById("caseBoxAll");
        totalCheck.checked = false;

        var boxs       = document.getElementsByName("script-boxs");
        for(i = 0; i < boxs.length; i++){
            boxs[i].checked = false;
        }
    }
</script>
