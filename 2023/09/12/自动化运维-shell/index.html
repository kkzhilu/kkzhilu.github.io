自动化运维 - shell命令

#!/bin/bash
#这里可替换为你自己的执行程序以及日志目录，其他代码无需更改
APP_NAME=juejin-news-1.0.0-SNAPSHOT.jar
PROPERTIES_PATH=application.properties

#使用说明，用来提示输入参数
usage() {
    echo "Usage: sh 执行脚本.sh [start|stop|restart|status]"
    exit 1
}

#检查程序是否在运行
is_exist(){
  pid=`ps -ef|grep $APP_NAME|grep -v grep|grep -v kill|awk '{print $2}' `
  #如果不存在返回1，存在返回0     
  if [ -z "${pid}" ]; then
   return 1
  else
    return 0
  fi
}

#启动方法
start(){
  is_exist
  if [ $? -eq "0" ]; then
    echo "${APP_NAME} is already running. pid=${pid} ."
  else
     nohup java -Xms1024m -Xmx1024m -XX:PermSize=128M -XX:MaxNewSize=256m -XX:MaxPermSize=256m -jar -Dspring.config.location=$PROPERTIES_PATH $APP_NAME  > catalina.out  2>&1 &
  fi
}

#停止方法
stop(){
  is_exist
  if [ $? -eq "0" ]; then
    kill -9 $pid
  else
    echo "${APP_NAME} is not running"
  fi
}

#输出运行状态
status(){
  is_exist
  if [ $? -eq "0" ]; then
    echo "${APP_NAME} is running. Pid is ${pid}"
  else
    echo "${APP_NAME} is NOT running."
  fi
}

#重启
restart(){
  stop
  start
}


#根据输入参数，选择执行对应方法，不输入则执行使用说明
case "$1" in
  "start")
    start
    ;;
  "stop")
    stop
    ;;
  "status")
    status
    ;;
  "restart")
    restart
    ;;
  *)
    usage
    ;;
esac

rpcapp(){
  DOCKER_ID=$(docker ps | grep "rpcapp" | awk '{print $1}')
  docker exec -it $DOCKER_ID bash

# 执行命令
mongo xxx.js
var mongoInfo = "127.0.0.1:27017/pmreal_db";

db = connect(mongoInfo);

print("connect database. database:" + mongoInfo);

// 需要调整的表
var tarray = new Array("pmonitorScriptRunInfo");

// 索引
var monitorFinishTimeIndexes = new Array({"monitorid": 1, "finishTime": -1});

for (var tindex = 0; tindex < tarray.length; tindex++) {

  var tablename = tarray[tindex];

  var tableIndexes = monitorFinishTimeIndexes;

  for (var i = 0; i < 100; i++) {
    var tname;
    if (i < 10) {
      tname = tablename + "_0" + i;
    } else {
      tname = tablename + "_" + i;
    }

    print("Building index. tablename:" + tname);

    // 建立索引
    buildIndex(db, tname, tableIndexes);
  }
}

// 建立索引
function buildIndex(db, tablename, tableIndexes) {
  for (var i = 0; i < tableIndexes.length; i++) {
    var tableIndex = tableIndexes[i];

    db[tablename].ensureIndex(tableIndex, {background:true});

    print("Build index success. tablename:" + tablename + "; tableindex:" + JSON.stringify(tableIndex));
  }
}
